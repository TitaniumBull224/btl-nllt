
            class main {
                func isPrime(a:int):bool {
                    if a < 2 {
                        return false;
                    }
                    for i := 2 ; i * i <= a; i := i + 1 {
                        if a % i == 0 {
                            return false;
                        }
                    }
                    return true;
                }
            }
