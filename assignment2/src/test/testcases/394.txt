
            class main {
                func fibo(n:int):int {
                    if n == 0 {
                        return 0;
                    }
                    if n == 1 {
                        return 1;
                    }
                    var a, b:int = 0, 1;
                    var c:int;
                    for i := 2; i <= n; i := i + 1 {
                        c := a + b;
                        a := b;
                        b := c;
                    }
                    return b;
                
                }
            }
